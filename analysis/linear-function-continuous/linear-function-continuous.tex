    \begin{theorem}[Linear function $\mathbb{R}^m \to \mathbb{R}$]
       A linear function $f  \colon \mathbb{R}^{m} \to \mathbb{R}$
       is uniformly continuous.
    \end{theorem}

    \begin{proof}
        Let $e_1, \ldots, e_m$ be the usual basis for $\mathbb{R}^{m}$.
        By linearity, 
        \begin{align*}
            \left| f(x) \right|^2 
            &= \left| x_1 f(e_1) + \ldots + 
            x_m f(e_m) \right|^2\\
            &\le \left( x_1^2 + \ldots + x_m^2 \right) 
            \left( f\left( e_1 \right)^2 + \ldots +
            f\left( e_m \right)^2 \right) \tag{Cauchy-Schwarz}
        \end{align*}
    \end{proof}
    So for all $x \in \mathbb{R}$, we have
    \[
    \left| f(x) \right| \le M \|x\|, \quad M
    = \left( f(e_1)^2 +\ldots + f(e_m)^2 \right)^{\frac{1}{2}}.
    \] 
    It follows that
    \[
    \left| f(x) - f(x_0) \right| = 
    \left| f\left( x-x_0 \right)  \right| 
    \le M \|x - x_0\|.
    \] 

    \begin{note}
        This is only a weak version. In fact, we have the following theorem:
        \begin{theorem}\cite[p~638]{Lee}
        A linear map between
        finite-dimensional normed linear spaces is continuous.
        \end{theorem}
    \end{note}
    
    \begin{corollary}[]
        Linear isomorphisms between finite-dimensional normed vector spaces are
        homeomorphisms.
    \end{corollary}
